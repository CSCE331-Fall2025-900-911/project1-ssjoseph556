def factorial(n):
    """
    Calculates the factorial of a non-negative integer.
    Generated by AI.
    """
    if n < 0:
        return "Factorial does not exist for negative numbers"
    elif n == 0:
        return 1
    else:
        result = 1
        for i in range(1, n + 1):
            result *= i
        return result
